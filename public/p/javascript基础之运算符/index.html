let a1 = 1
let b1 = 2
let c1 = b1 + a1++
console.log(a1, b1, c1);//2,2,3

let a2 = 1
let b2 = 2
let c2 = b2 + ++a2
console.log(a2, b2, c2);//2,2,4

说明：

  • 后置是先计算再++
  • 前置是先++再计算
